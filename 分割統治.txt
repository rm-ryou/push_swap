
bool	push_under10(t_info *info, int pa_num)
{
	pa_num = 0;
	t_dlist	*cur;
	int	mid_a = 0;
	int	mid_b = dlist_size(info->b) / 2 + 1;
	if (mid_b <= 3)
		mid_b = 3;
	if (dlist_size(info->b) < 3)
		return (true);
	while (dlist_size(info->b) != mid_b -1)
	{
		cur = info->b->next;
		if (cur->index >= mid_b)
		{
			pa(info);
			pa_num += 1;
		}
		else
			rb(info);
	}
	push_under10(info, pa_num);
	check_swap(info, 2);
	push_b(info, pa_num);

}
